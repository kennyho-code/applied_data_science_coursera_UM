An NOAA dataset has been stored in the file data/C2A2_data/BinnedCsvs_d400/6f3ea56a5c5374c66522d5595cf0e7ed5fbbe9ac0e7a2c5fd5cc3cb5.csv. The data for this assignment comes from a subset of The National Centers for Environmental Information (NCEI) Daily Global Historical Climatology Network (GHCN-Daily). The GHCN-Daily is comprised of daily climate records from thousands of land surface stations across the globe.

Each row in the assignment datafile corresponds to a single observation.

The following variables are provided to you:

  • id : station identification code
  • date : date in YYYY-MM-DD format (e.g. 2012-01-24 = January 24, 2012)
  • element : indicator of element type
    • TMAX : Maximum temperature (tenths of degrees C)
    • TMIN : Minimum temperature (tenths of degrees C)
  • value : data value for element (tenths of degrees C)

For this assignment, you must:

  1. Read the documentation and familiarize yourself with the dataset, then write some python code which returns a line graph of the record high and record low temperatures by day of the year over the period 2005-2014. The area between the record high and record low temperatures for each day should be shaded.
  2. Overlay a scatter of the 2015 data for any points (highs and lows) for which the ten year record (2005-2014) record high or record low was broken in 2015.
  3. Watch out for leap days (i.e. February 29th), it is reasonable to remove these points from the dataset for the purpose of this visualization.
  4. Make the visual nice! Leverage principles from the first module in this course when developing your solution. Consider issues such as legends, labels, and chart junk.

import matplotlib as mpl
mpl.get_backend()
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
import pandas as pd
from matplotlib.patches import Rectangle
from matplotlib.lines import Line2D   

def celsius2fahrenheit(temp):
    """
    Returns temperature in Celsius.
    """
    return (temp) * (9/5) + 32

def create_max_array(arr, max_arr):
    for i in range(len(max_arr)):
        if arr[i] < max_arr[i]:
            arr[i] = None

def create_min_array(arr, min_arr):
    for i in range(len(min_arr)):
        if arr[i] >= min_arr[i]:
            arr[i] = None

    
def show_project(binsize, hashid):
    # read csv and create dataframe
    df = pd.read_csv('data/C2A2_data/BinSize_d{}.csv'.format(binsize))
    station_locations_by_hash = df[df['hash'] == hashid]
    df_temps = pd.read_csv('data/C2A2_data/BinnedCsvs_d400/6f3ea56a5c5374c66522d5595cf0e7ed5fbbe9ac0e7a2c5fd5cc3cb5.csv')
    df_temps['Date'] = pd.to_datetime(df_temps['Date'])
    
    
    #*** 2000 - 2014 ***
    df_00_14 = df_temps[(df_temps['Date'] >= '2000-01-01') & (df_temps['Date'] <= '2014-12-31')]
    df_00_14 = df_00_14.groupby([df_00_14.Date.dt.month, df_00_14.Date.dt.day])
    #max_00_14 = df_00_14.aggregate(np.max)
    max_00_14 = list(df_00_14['Data_Value'].max())
    max_00_14 = list(map(lambda x: x / 10, max_00_14))
    # delete leap year
    del max_00_14[59]
    min_00_14 = list(df_00_14['Data_Value'].min())
    min_00_14 = list(map(lambda x: x / 10, min_00_14))
    del min_00_14[59]
    
    
    #*** 2015 ***
    df_15 = df_temps[(df_temps['Date'] >= '2015-01-01') & (df_temps['Date'] <= '2015-12-31')]
    df_15 = df_15.groupby([df_15.Date.dt.month, df_15.Date.dt.day])
    max_15 = df_15['Data_Value'].max()
    max_15 = list(map(lambda x: x / 10, max_15))
    # make max_15 only show maxes
    create_max_array(max_15, max_00_14)
    min_15 = df_15['Data_Value'].min()
    min_15 = list(map(lambda x: x / 10, min_15))
    # make min_15 only show mins
    create_min_array(min_15, min_00_14)
    

    # plot and options 
    f, ax_f = plt.subplots()
    f.suptitle('Maximum and Minumum Temperatures from 2005 to 2014 with 2015 Points in Denver, Co', x=.51,color='#566573', fontsize=10)
    plt.xlabel('Months', fontsize=10, labelpad=10, color='#566573')
    plt.ylabel('$^\circ$ C', fontsize=10, labelpad=10, color='#566573')

    
    # plot lines
    plt.plot(max_00_14, '-', color='#FADBD8')
    plt.plot(min_00_14, '-', color='#D6EAF8')
    
    line1, = plt.plot([1,2,3], linestyle='-', color='#FADBD8')
    line2, = plt.plot([1,2,3], linestyle='-', color='#D6EAF8')
    dot1 = Line2D(range(1), range(1), color="white", marker='o', markerfacecolor='#CB4335')
    dot2 = Line2D(range(1), range(1), color="white", marker='o', markerfacecolor='#2874A6')

    legend = plt.legend([line1, line2, dot1, dot2], ['Max 2000-2014', 'Min 2000-2014', 'Max 2015', 'Min 2015'], loc='upper right')
    l = legend
    for text in l.get_texts():
        text.set_color('#566573')
        text.set_size(6)

    
    plt.plot(max_15, '.', color='#CB4335')
    plt.plot(min_15, '.', color='#2874A6')

    
    # right y axis 
    y1, y2 = ax_f.get_ylim()
    y2 += 10
    ax_c = ax_f.twinx()
    ax_c.set_ylim(celsius2fahrenheit(y1), celsius2fahrenheit(y2))
    
    plt.ylabel('$^\circ$ F', color='#566573', labelpad=10)
    
    ax_f.fill_between(np.arange(0,365), min_00_14, max_00_14, facecolor='#F2F3F4')
    
    
    # spacing for x-axis months 
    plt.xticks(np.arange(0, 365, 33.18))
    ax_f.set_xticklabels(['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'], fontsize = 10, ha='center', color="#566573")


    for spine in ax_f.spines.values():
        spine.set_color('#E5E7E9')
        
    for spine in ax_c.spines.values():
        spine.set_color('#E5E7E9')
        
    
    ax_f.spines['top'].set_visible(False)
    ax_c.spines['top'].set_visible(False)
        
    
    ax_f.tick_params(color='#E5E7E9')
    ax_c.tick_params(color='#E5E7E9')
    
    ax_f.tick_params(axis='y', colors='#566573')
    ax_c.tick_params(axis='y', colors='#566573')
    

    f.set_figheight(6)
    f.set_figwidth(9)
    
    plt.show()
    
    
show_project(400,'6f3ea56a5c5374c66522d5595cf0e7ed5fbbe9ac0e7a2c5fd5cc3cb5')
